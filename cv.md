```
function isIsogram (str) {
	str = str.toUpperCase();
	for ( let i = 0; i < str.length; i++) {
		for ( let b = i + 1; b < str.length; b++) {
			if (str[i] == str[b]) {
			return false;
    			}
    		}
    	}
    	return true;
}
```
